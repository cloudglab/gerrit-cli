gerrit-cli

Gerrit 命令行工具 · bun install -g @cloudglab/gerrit-cli

安装 & 升级

# 安装 Bun 运行时
curl -fsSL https://bun.sh/install | bash

# 全局安装
bun install -g @cloudglab/gerrit-cli

# 升级
bun update -g @cloudglab/gerrit-cli

环境变量

export GERRIT_HOST="https://gerrit.example.com"
export GERRIT_USERNAME="your-username"
export GERRIT_PASSWORD="your-http-password"

或运行 gerrit-cli setup 写入配置文件。

push mine show rebase checkout

场景速查

提交变更

git commit -m "feat: add login"
gerrit-cli push -t my-feature -r alice@example.com

审查

gerrit-cli incoming                                   # 待审查列表
gerrit-cli show 12345                                 # 变更详情
gerrit-cli diff 12345                                 # 查看 diff
gerrit-cli comments 12345                             # 已有评论
gerrit-cli comment 12345 -m "LGTM, one nit"           # 发布评论
gerrit-cli vote 12345 --code-review 2 -m "Approved"   # 投票

CI 状态

gerrit-cli build-status 12345                        # 单次快照
gerrit-cli build-status --watch --exit-status         # 轮询等待
gerrit-cli extract-url "jenkins" | tail -1            # 提取构建链接

工作区

gerrit-cli checkout 12345                            # 检出变更
gerrit-cli workspace create 12345                     # 创建工作树
gerrit-cli push -b main -t refactor --wip             # 推送 WIP

变更管理

gerrit-cli rebase 12345                              # 重基
gerrit-cli submit 12345                               # 合入
gerrit-cli abandon 12345                              # 丢弃
gerrit-cli restore 12345                              # 恢复

组管理

gerrit-cli groups                                    # 组列表
gerrit-cli groups-show administrators                 # 组详情
gerrit-cli groups-members project-reviewers           # 组成员

批量管道

# 批量行评论
echo '[{"file":"src/app.ts","line":42,"message":"Use const"}]' \
  | gerrit-cli comment 12345 --batch

# 批量添加审查人
gerrit-cli add-reviewer alice bob -c 12345

搜索

gerrit-cli search "owner:self status:open"
gerrit-cli search "project:my-project" -n 10
gerrit-cli search "status:merged age:7d"

配置 & 辅助

gerrit-cli setup                                     # 初始化
gerrit-cli status                                    # 连接状态
gerrit-cli config show                               # 显示配置
gerrit-cli version                                   # 版本号
gerrit-cli completion bash                           # 生成补全脚本
gerrit-cli clean --dry-run                           # 清理已合并分支
